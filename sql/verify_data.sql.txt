-- Total Records

SELECT COUNT(*)
FROM ecommerce_dw.dim_customer;

-- Preview Data

SELECT *
FROM ecommerce_dw.dim_customer
LIMIT 20;

-- Customer-wise Spending

SELECT
    customer_name,
    total_spent
FROM ecommerce_dw.dim_customer
ORDER BY total_spent DESC;

-- Active Customers

SELECT *
FROM ecommerce_dw.dim_customer
WHERE account_status = 'Active';
